184  SUBROUTINE oasis_var_setup()
185  IMPLICIT NONE
186 
187  !--------------------------------------------------------
188  integer(kind=ip_intwp_p) :: m,n,p,v
189  INTEGER(kind=ip_intwp_p) :: ierr, taskid
190  integer(kind=ip_intwp_p) :: vcnt
191  logical :: found, fastcheckout
192  character(len=ic_lvar) ,pointer :: vname0(:),vname(:)
193  character(len=ic_lvar2) ,pointer :: pname0(:),pname(:)
194  integer(kind=ip_intwp_p),pointer :: inout0(:),inout(:)
195  integer(kind=ip_intwp_p),pointer :: vanum0(:),vanum(:)
196  logical, parameter :: local_timers_on = .false.
197  character(len=*),parameter :: subname = '(oasis_var_setup)'
198  !--------------------------------------------------------
199 
200  call oasis_debug_enter(subname)
201 
202  IF (local_timers_on) call oasis_timer_start('var_setup')
203 
204  IF (local_timers_on) call oasis_timer_start('var_setup_reducelists')
205  allocate(vname0(prism_nvar))
206  allocate(pname0(prism_nvar))
207  allocate(inout0(prism_nvar))
208  allocate(vanum0(prism_nvar))
209  do n = 1,prism_nvar
210  vname0(n) = prism_var(n)%name
211  inout0(n) = prism_var(n)%ops
212  vanum0(n) = prism_var(n)%num
213  pname0(n) = prism_part(prism_var(n)%part)%partname
214  enddo
215 
216  call oasis_mpi_reducelists(vname0,mpi_comm_local,vcnt,vname,'var_setup', &
217  fastcheck=.true.,fastcheckout=fastcheckout, &
218  linp2=pname0,lout2=pname,linp3=inout0,lout3=inout,linp4=vanum0,lout4=vanum)
219 
220  deallocate(vname0)
221  deallocate(pname0)
222  deallocate(inout0)
223  deallocate(vanum0)
224  IF (local_timers_on) call oasis_timer_stop('var_setup_reducelists')
225 
226  !-------------------------------------------------
227  !> * Initialize variables on tasks where they are not previously defined.
228  ! if fastcheck worked, then don't need to do this extra work to add undefined vars
229  !-------------------------------------------------
230 
231  if (.not. fastcheckout) then
232 
233  if (local_timers_on) call oasis_timer_start('var_setup_initvar')
234  do v = 1,vcnt
235 
236  !--- either a prism_var that already exists
237  found = .false.
238  n = 0
239  do while (n < prism_nvar .and. .not.found)
240  n = n + 1
241  if (prism_var(n)%name == vname(v)) then
242  found = .true.
243  endif
244  enddo
245 
246  !--- or a new prism_var that must be instantiated
247  if (.not.found) then
248  prism_nvar = prism_nvar + 1
249 
250  call oasis_var_zero(prism_var(prism_nvar))
251  prism_var(prism_nvar)%name = vname(v)
252  prism_var(prism_nvar)%ops = inout(v)
253  prism_var(prism_nvar)%num = vanum(v)
254  prism_var(prism_nvar)%ncpl = 0
255  !--- figure out the local part id for the part name
256  p = 0
257  found = .false.
258  do while (p < prism_npart .and. .not.found)
259  p = p + 1
260  if (prism_part(p)%partname == pname(v)) then
261  found = .true.
262  endif
263  enddo
264  if (found) then
265  prism_var(prism_nvar)%part = p
266  if (oasis_debug >= 15) then
267  write(nulprt,*) subname,' found part match ',trim(vname(v)),trim(pname(v)),p
268  endif
269  else
270  write(nulprt,*) subname,estr,'prism part not found part = ',trim(pname(v)),' var = ',trim(vname(v))
271  call oasis_abort(file=__file__,line=__line__)
272  endif
273 
274  if (oasis_debug >= 2) then
275  write(nulprt,*) ' '
276  write(nulprt,*) subname,' add var = ',prism_nvar,trim(prism_var(prism_nvar)%name),&
277  prism_var(prism_nvar)%part,&
278  trim(prism_part(prism_var(prism_nvar)%part)%partname),prism_var(prism_nvar)%ops
279  CALL oasis_flush(nulprt)
280  ENDIF
281  endif
282 
283  enddo ! v = 1,vcnt
284  if (local_timers_on) call oasis_timer_stop ('var_setup_initvar')
285 
286  endif ! fastcheckout
287 
288  deallocate(vname,pname,inout,vanum)
289 
290  IF (local_timers_on) call oasis_timer_stop('var_setup')
291 
292  call oasis_debug_exit(subname)
293 
294  END SUBROUTINE oasis_var_setup
